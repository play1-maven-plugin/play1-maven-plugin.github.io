105     public PlayJUnit4Provider( ProviderParameters bootParams )
106     {
107         // don't start a thread in CommandReader while we are in in-plugin process
108         commandsReader = bootParams.isInsideFork() ? getReader().setShutdown( bootParams.getShutdown() ) : null;
109         providerParameters = bootParams;
110         testClassLoader = bootParams.getTestClassLoader();
111         consoleLogger = bootParams.getConsoleLogger();
112         //scanResult = bootParams.getScanResult();
113         scanResult = PlayScanResult.from( bootParams.getProviderProperties(), consoleLogger );
114         runOrderCalculator = bootParams.getRunOrderCalculator();
115         String listeners = bootParams.getProviderProperties().get( "listener" );
116         customRunListeners = unmodifiableCollection( createCustomListeners( listeners ) );
117         jUnit4TestChecker = new JUnit4TestChecker( testClassLoader );
118         TestRequest testRequest = bootParams.getTestRequest();
119         testResolver = testRequest.getTestListResolver();
120         rerunFailingTestsCount = testRequest.getRerunFailingTestsCount();
121     }

135     // Copy of AbstractPlayMojo.getPlayHome() method (with getCanonicalPath() changed to getAbsolutePath() )
136     private File getPlayHome( File applicationPath )
137         throws TestSetFailedException
138     {
139         File targetDir = new File( applicationPath, "target" );
140         File playTmpDir = new File( targetDir, "play" );
141         File playTmpHomeDir = new File( playTmpDir, "home" );
142         if ( !playTmpHomeDir.exists() )
143         {
144             throw new TestSetFailedException( String.format( "Play! home directory \"%s\" does not exist",
145                                                        playTmpHomeDir.getAbsolutePath() ) );
146         }
147         if ( !playTmpHomeDir.isDirectory() )
148         {
149             throw new TestSetFailedException( String.format( "Play! home directory \"%s\" is not a directory",
150                                                        playTmpHomeDir.getAbsolutePath() ) );
151         }
152         // Additional check whether the temporary Play! home directory is created by this plugin
153         File warningFile = new File( playTmpHomeDir, "WARNING.txt" );
154         if ( warningFile.exists() )
155         {
156             if ( !warningFile.isFile() )
157             {
158                 throw new TestSetFailedException( String.format( "Play! home directory warning file \"%s\" is not a file",
159                                                            warningFile.getAbsolutePath() ) );
160             }
161         }
162         else
163         {
164             throw new TestSetFailedException( String.format( "Play! home directory warning file \"%s\" does not exist",
165                                                        warningFile.getAbsolutePath() ) );
166         }
167         return playTmpHomeDir;
168     }

170     private File checkPath( String path ) throws TestSetFailedException
171     {
172         if ( path == null )
173         {
174             throw new TestSetFailedException( "Path is null" );
175         }
176         File file = new File( path );
177         if ( !file.exists() )
178         {
179             throw new TestSetFailedException( "Path \"" + path + "\" does not exist" );
180         }
181         if ( !file.isDirectory() )
182         {
183             throw new TestSetFailedException( "Path \"" + path + "\" is not a directory" );
184         }
185         return file;
186     }

188     public RunResult invoke( Object forkTestSet )
189         throws TestSetFailedException
190     {
191         consoleLogger.info( "Play! initialization\n" );
192 
193         initializePlayEngine();
194         try
195         {
196             upgradeCheck();
197 
198             ReporterFactory reporterFactory = providerParameters.getReporterFactory();
199 
200             RunResult runResult;
201             try
202             {
203                 RunListener reporter = reporterFactory.createReporter();
204 
205                 startCapture( (ConsoleOutputReceiver) reporter );
206                 // startCapture() called in prior to setTestsToRun()
207 
208                 if ( testsToRun == null )
209                 {
210                     setTestsToRun( forkTestSet );
211                 }
212 
213                 Notifier notifier = new Notifier( new JUnit4RunListener( reporter ), getSkipAfterFailureCount() );
214                 Result result = new Result();
215                 notifier.addListeners( customRunListeners )
216                     .addListener( result.createListener() );
217 
218                 if ( isFailFast() && commandsReader != null )
219                 {
220                     registerPleaseStopJUnitListener( notifier );
221                 }
222 
223                 try
224                 {
225                     notifier.fireTestRunStarted( testsToRun.allowEagerReading()
226                                                      ? createTestsDescription( testsToRun )
227                                                      : createDescription( UNDETERMINED_TESTS_DESCRIPTION ) );
228 
229                     if ( commandsReader != null )
230                     {
231                         registerShutdownListener( testsToRun );
232                         commandsReader.awaitStarted();
233                     }
234 
235                     for ( Class<?> testToRun : testsToRun )
236                     {
237                         executeTestSet( testToRun, reporter, notifier );
238                     }
239                 }
240                 finally
241                 {
242                     notifier.fireTestRunFinished( result );
243                     notifier.removeListeners();
244                 }
245                 rethrowAnyTestMechanismFailures( result );
246             }
247             finally
248             {
249                 runResult = reporterFactory.close();
250             }
251             return runResult;
252         }
253         finally
254         {
255             consoleLogger.info( "Play! finalization\n" );
256             finalizePlayEngine();
257         }
258     }

260     private void initializePlayEngine()
261         throws TestSetFailedException
262     {
263         File applicationPath = getApplicationPath();
264         File playHome = getPlayHome( applicationPath );
265         String playId = getProviderProperty( "play.testId", "test" );
266         int playStartTimeout = Integer.parseInt( getProviderProperty( "play.startTimeout", "0" ) );
267         runRunnable( new PlayStartRunnable( playHome, applicationPath, playId ), "Play! initialization",
268                        playStartTimeout );
269     }
270 
271     private void finalizePlayEngine()
272         throws TestSetFailedException
273     {
274         int playStopTimeout = Integer.parseInt( getProviderProperty( "play.stopTimeout", "0" ) );
275         runRunnable( new PlayStopRunnable(), "Play! finalization", playStopTimeout );
276     }

278     private void runRunnable( AbstractRunnable runnable, String threadName, int timeout )
279         throws TestSetFailedException
280     {
281         if ( timeout <= 0 )
282         {
283             try
284             {
285                 runnable.methodToRun();
286             }
287             catch ( Exception e )
288             {
289                 throw new TestSetFailedException( threadName + " error", e );
290             }
291         }
292         else
293         {
294             Thread t = new Thread( runnable, threadName );
295             t.setDaemon( true );
296             t.start();
297             try
298             {
299                 t.join( timeout );
300             }
301             catch ( InterruptedException e )
302             {
303                 t.interrupt();
304                 throw new TestSetFailedException( threadName + " interrupted", e );
305             }
306             Exception runnerException = runnable.getException();
307             if ( runnerException != null )
308             {
309                 // If there is an exception, the thread is not alive anymore t.interrupt();
310                 throw new TestSetFailedException( threadName + " error", runnerException );
311             }
312             if ( !runnable.isExecuted() ) // Thread still alive
313             {
314                 t.interrupt();
315                 throw new TestSetFailedException( threadName + " timed out" );
316             }
317         }
318     }

320     private void setTestsToRun( Object forkTestSet )
321         throws TestSetFailedException
322     {
323         if ( forkTestSet instanceof TestsToRun )
324         {
325             testsToRun = (TestsToRun) forkTestSet;
326         }
327         else if ( forkTestSet instanceof Class )
328         {
329             testsToRun = fromClass( (Class<?>) forkTestSet );
330         }
331         else
332         {
333             testsToRun = scanClassPath();
334         }
335     }

352     private void registerShutdownListener( final TestsToRun testsToRun )
353     {
354         commandsReader.addShutdownListener( new CommandListener()
355         {
356             public void update( Command command )
357             {
358                 testsToRun.markTestSetFinished();
359             }

363     private void registerPleaseStopJUnitListener( final Notifier notifier )
364     {
365         commandsReader.addSkipNextTestsListener( new CommandListener()
366         {
367             public void update( Command command )
368             {
369                 notifier.pleaseStop();
370             }
371         } );

374     private void executeTestSet( Class<?> clazz, RunListener reporter, Notifier notifier )
375     {
376         final ReportEntry report = new SimpleReportEntry( getClass().getName(), clazz.getName() );
377         reporter.testSetStarting( report );
378         try
379         {
380             executeWithRerun( clazz, notifier );
381         }
382         catch ( Throwable e )
383         {
384             if ( isFailFast() && e instanceof StoppedByUserException )
385             {
386                 String reason = e.getClass().getName();
387                 Description skippedTest = createDescription( clazz.getName(), createIgnored( reason ) );
388                 notifier.fireTestIgnored( skippedTest );
389             }
390             else
391             {
392                 String reportName = report.getName();
393                 String reportSourceName = report.getSourceName();
394                 PojoStackTraceWriter stackWriter = new PojoStackTraceWriter( reportSourceName, reportName, e );
395                 reporter.testError( withException( reportSourceName, reportName, stackWriter ) );
396             }
397         }
398         finally
399         {
400             reporter.testSetCompleted( report );
401         }
402     }

404     private void executeWithRerun( Class<?> clazz, Notifier notifier )
405         throws TestSetFailedException
406     {
407         JUnitTestFailureListener failureListener = new JUnitTestFailureListener();
408         notifier.addListener( failureListener );
409         boolean hasMethodFilter = testResolver != null && testResolver.hasMethodPatterns();
410 
411         try
412         {
413             try
414             {
415                 notifier.asFailFast( isFailFast() );
416                 execute( clazz, notifier, hasMethodFilter ? createMethodFilter() : null );
417             }
418             finally
419             {
420                 notifier.asFailFast( false );
421             }
422 
423             // Rerun failing tests if rerunFailingTestsCount is larger than 0
424             if ( isRerunFailingTests() )
425             {
426                 Notifier rerunNotifier = pureNotifier();
427                 notifier.copyListenersTo( rerunNotifier );
428                 for ( int i = 0; i < rerunFailingTestsCount && !failureListener.getAllFailures().isEmpty(); i++ )
429                 {
430                     Set<ClassMethod> failedTests = generateFailingTests( failureListener.getAllFailures() );
431                     failureListener.reset();
432                     if ( !failedTests.isEmpty() )
433                     {
434                         executeFailedMethod( rerunNotifier, failedTests );
435                     }
436                 }
437             }
438         }
439         finally
440         {
441             notifier.removeListener( failureListener );
442         }
443     }

451     private TestsToRun scanClassPath()
452     {
453         final TestsToRun scannedClasses = scanResult.applyFilter( jUnit4TestChecker, Play.classloader/*testClassLoader*/ );
454         return runOrderCalculator.orderTestClasses( scannedClasses );
455     }

457     private void upgradeCheck()
458         throws TestSetFailedException
459     {
460         if ( isJUnit4UpgradeCheck() )
461         {
462             Collection<Class<?>> classesSkippedByValidation =
463                 scanResult.getClassesSkippedByValidation( jUnit4TestChecker, testClassLoader );
464             if ( !classesSkippedByValidation.isEmpty() )
465             {
466                 StringBuilder reason = new StringBuilder();
467                 reason.append( "Updated check failed\n" );
468                 reason.append( "There are tests that would be run with junit4 / surefire 2.6 but not with [2.7,):\n" );
469                 for ( Class<?> testClass : classesSkippedByValidation )
470                 {
471                     reason.append( "   " );
472                     reason.append( testClass.getName() );
473                     reason.append( "\n" );
474                 }
475                 throw new TestSetFailedException( reason.toString() );
476             }
477         }
478     }

480     static Description createTestsDescription( Iterable<Class<?>> classes )
481     {
482         // "null" string rather than null; otherwise NPE in junit:4.0
483         Description description = createDescription( "null" );
484         for ( Class<?> clazz : classes )
485         {
486             description.addChild( createDescription( clazz.getName() ) );
487         }
488         return description;
489     }

491     private static boolean isJUnit4UpgradeCheck()
492     {
493         return System.getProperty( "surefire.junit4.upgradecheck" ) != null;
494     }

496     private static void execute( Class<?> testClass, Notifier notifier, Filter filter )
497         throws TestSetFailedException
498     {
499         final int classModifiers = testClass.getModifiers();
500         if ( !isAbstract( classModifiers ) && !isInterface( classModifiers ) )
501         {
502             Request request = aClass( testClass );
503             if ( filter != null )
504             {
505                 request = request.filterWith( filter );
506             }
507             Runner runner = request.getRunner();
508             if ( countTestsInRunner( runner.getDescription() ) != 0 )
509             {
510                 // runner.run( notifier );
511                 executeInPlayContext( runner, notifier );
512             }
513         }
514     }
515 
516     private void executeFailedMethod( RunNotifier notifier, Set<ClassMethod> failedMethods )
517         throws TestSetFailedException
518     {
519         for ( ClassMethod failedMethod : failedMethods )
520         {
521             try
522             {
523                 Class<?> methodClass = Class.forName( failedMethod.getClazz(), true, testClassLoader );
524                 String methodName = failedMethod.getMethod();
525                 // method( methodClass, methodName ).getRunner().run( notifier );
526                 Runner runner = method( methodClass, methodName ).getRunner();
527                 executeInPlayContext( runner, notifier );
528             }
529             catch ( ClassNotFoundException e )
530             {
531                 throw new TestSetFailedException( "Unable to create test class '" + failedMethod.getClazz() + "'", e );
532             }
533         }
534     }

536     /**
537      * JUnit error: test count includes one test-class as a suite which has filtered out all children.
538      * Then the child test has a description "initializationError0(org.junit.runner.manipulation.Filter)"
539      * for JUnit 4.0 or "initializationError(org.junit.runner.manipulation.Filter)" for JUnit 4.12
540      * and Description#isTest() returns true, but this description is not a real test
541      * and therefore it should not be included in the entire test count.
542      */
543     private static int countTestsInRunner( Description description )
544     {
545         if ( description.isSuite() )
546         {
547             int count = 0;
548             for ( Description child : description.getChildren() )
549             {
550                 if ( !hasFilteredOutAllChildren( child ) )
551                 {
552                     count += countTestsInRunner( child );
553                 }
554             }
555             return count;
556         }
557         else if ( description.isTest() )
558         {
559             return hasFilteredOutAllChildren( description ) ? 0 : 1;
560         }
561         else
562         {
563             return 0;
564         }
565     }
566 
567     private static boolean hasFilteredOutAllChildren( Description description )
568     {
569         String name = description.getDisplayName();
570         // JUnit 4.0: initializationError0; JUnit 4.12: initializationError.
571         if ( name == null )
572         {
573             return true;
574         }
575         else
576         {
577             name = name.trim();
578             return name.startsWith( "initializationError0(org.junit.runner.manipulation.Filter)" )
579                            || name.startsWith( "initializationError(org.junit.runner.manipulation.Filter)" );
580         }
581     }

583     private Filter createMethodFilter()
584     {
585         TestListResolver methodFilter = optionallyWildcardFilter( testResolver );
586         return methodFilter.isEmpty() || methodFilter.isWildcard() ? null : new TestResolverFilter( methodFilter );
587     }

591     private static void executeInPlayContext( Runner runner, RunNotifier notifier )
592         throws TestSetFailedException
593     {
594         try
595         {
596             String invocationClassName = "com.google.code.play.surefire.junit4.TestInvocation";
597             if ( "1.2".compareTo( Play.version ) <= 0 )
598             {
599                 invocationClassName = "com.google.code.play.surefire.junit4.Play12TestInvocation";
600             }
601             Invoker.DirectInvocation invocation = getInvocation( invocationClassName, runner, notifier );
602             Invoker.invokeInThread( invocation );
603         }
604         catch ( Throwable e )
605         {
606             throw new TestSetFailedException( e );
607         }
608     }
609 
610     private static Invoker.DirectInvocation getInvocation( String invocationClassName, Runner runner, RunNotifier notifier )
611         throws Throwable
612     {
613         Invoker.DirectInvocation invocation = null;
614         Class<?> cl = Class.forName( invocationClassName );
615         Constructor<?> c = cl.getConstructor( Runner.class, RunNotifier.class );
616         invocation = (Invoker.DirectInvocation) c.newInstance( runner, notifier );
617         return invocation;
618     }
